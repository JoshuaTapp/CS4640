package com.example.assign2;

import org.junit.jupiter.api.AfterEach;
import org.junit.jupiter.api.Assertions;
import org.junit.jupiter.api.BeforeEach;
import org.junit.jupiter.api.Test;

import java.util.ArrayList;


public class BoundedQueueTest {

    @Test
    public void testConstructorIllegal() {
        IllegalArgumentException throwm = Assertions.assertThrows(IllegalArgumentException.class, ()-> {
            BoundedQueue q = new BoundedQueue(-1);
        }, "BoundedQueue.constructor");
    }

    @Test
    public void testConstructorLegal() {
        BoundedQueue q = new BoundedQueue(5);
        Assertions.assertEquals(true, q.isEmpty());
    }

    @Test
    public void testNullAdd() {
        BoundedQueue q = new BoundedQueue(5);

        NullPointerException throwm = Assertions.assertThrows(NullPointerException.class, ()-> {
            q.enQueue(null);
        }, "BoundedQueue.enQueue");
    }

    @Test
    public void testFullAdd() {
        BoundedQueue q = new BoundedQueue(0);

        IllegalStateException throwm = Assertions.assertThrows(IllegalStateException.class, ()-> {
            q.enQueue(new Integer[0]);
        }, "BoundedQueue.enQueue");
    }

    @Test
    public void testEnQueue() {
        BoundedQueue q = new BoundedQueue(3);
        for (int i = 0; i < 3; i++) {
            q.enQueue(new ArrayList<BoundedQueue>());
        }
        Assertions.assertEquals(true, q.isFull());
    }

    @Test
    public void testIllegalDeQueue() {
        BoundedQueue q = new BoundedQueue(3);
        for (int i = 0; i < 3; i++) {
            q.enQueue(new ArrayList<BoundedQueue>());
        }
        q.deQueue();
        q.deQueue();
        q.deQueue();

        IllegalStateException throwm = Assertions.assertThrows(IllegalStateException.class, ()-> {
            q.deQueue();
        }, "BoundedQueue.deQueue");
    }

    @Test
    public void testLegalDeQueue() {
        BoundedQueue q = new BoundedQueue(3);
        for (int i = 0; i < 3; i++) {
            q.enQueue(new ArrayList<BoundedQueue>());
        }
        Assertions.assertEquals(true, q.isFull());
        q.deQueue();
        Assertions.assertEquals(false, q.isFull());
        Assertions.assertEquals(false, q.isEmpty());
        q.deQueue();
        Assertions.assertEquals(false, q.isEmpty());
        Assertions.assertEquals(false, q.isFull());
        q.deQueue();
        Assertions.assertEquals(true, q.isEmpty());
        Assertions.assertEquals(false, q.isFull());
    }

    @Test
    public void testIsEmpty() {
        BoundedQueue q = new BoundedQueue(3);
        Assertions.assertTrue(q.isEmpty());
        for (int i = 0; i < 3; i++) {
            q.enQueue(new ArrayList<BoundedQueue>());
            Assertions.assertFalse(q.isEmpty());
        }
        q.deQueue();
        q.deQueue();
        q.deQueue();
        Assertions.assertTrue(q.isEmpty());
    }

    @Test
    public void testIsFull() {
        BoundedQueue q = new BoundedQueue(3);
        Assertions.assertFalse(q.isFull());
        for (int i = 0; i < 2; i++) {
            q.enQueue(new ArrayList<BoundedQueue>());
            Assertions.assertFalse(q.isFull());
        }
        q.enQueue(new ArrayList<BoundedQueue>());
        Assertions.assertTrue(q.isFull());
        q.deQueue();
        Assertions.assertFalse(q.isFull());
    }

    @Test
    public void testToString() {
        String result = "[Help, Me, Please]";
        BoundedQueue q = new BoundedQueue(3);
        q.enQueue(new String("Help"));
        q.enQueue(new String("Me"));
        q.enQueue(new String("Please"));
        Assertions.assertEquals(result, q.toString());
    }

    @Test
    public void testDeQueueOrdering() {
        BoundedQueue q = new BoundedQueue(3);
        String resultArr[] = {"Help", "Me", "Please"};
        q.enQueue(new String("Help"));
        q.enQueue(new String("Me"));
        q.enQueue(new String("Please"));
        Assertions.assertEquals(resultArr[0], q.deQueue());
        Assertions.assertEquals(resultArr[1], q.deQueue());
        Assertions.assertEquals(resultArr[2], q.deQueue());
    }
}

public class PrimeNumbers implements Iterable<Integer>

   public void computePrimes(int n)
   {
      int count = 1;        // count of primes
      int number = 2;       // number tested for primeness
      boolean isPrime;      // is this number a prime

      while (count <= n)
      {
         isPrime = true;
         for (int divisor = 2; divisor <= number / 2; divisor++)
         {
            if (number % divisor == 0)
            {
               isPrime = false;
               break;      // for loop
            }
         }
         if (isPrime && (number % 10 != 9))        // FAULT
         {
            primes.add(number);
            count++; 
         }
         number++; 
      }
   }
   
   @Override public Iterator<Integer> iterator()

      return primes.iterator();
   }
    
   @Override public String toString()

   }

computePrimes() has a fault that causes it not to include prime numbers whose last digit is 9 (for example, it omits 19, 29, 59, 79, 89, 109, ...).

If possible, describe the following five tests. You can describe the tests as sequences of calls to the above methods, or briefly describe them in words. Note that the last two tests require the test oracle to be described. If a test cannot be created, explain why.

  1. (2 points) A test that does not reach the fault
    1. Input = n=0, Output = []
    1. Because 0 < count, the while loop is skipped and does not touch the fault.
  1. (2 points) A test that reaches the fault, but does not infect
    1. Input = n=1 , Output = [2]
    1. We know the first number 2 is prime. It will skip the first if conditional and enter the fault. This is because isPrime was not modified and 2 % 10 = 2 which is not equal to 9. This is the expected behavior and even though we have the fault, the program does not encounter an error.
    1. Further, any input where 0 < n < 8 will reach the fault, but not infect.
  1. (2 points) A test that infects the state, but does not propagate
    1. I do not believe this is possible in this instance. The only time we want to enter the fault and do not by error, is when we have a prime number that ends in 9. When this happens, we skip it and continue to look for prime numbers. The smallest case is when n=8 because the 8th prime is 19. In this case, our output is: [2, 3, 5, 7, 11, 13, 17, 23]. This internal state always propagates to the final output, because the internal state and final state are tightly coupled. The fault causes the count to have a value -1 of its correct value. This leads to the final output always having the incorrect values.
  1. (2 points) A test that propagates, but does not reveal
    1. If we imagine our test oracle as something like printing the program states during each loop as the “observed portion of the program state” and our final print out as the incorrect portion of the final state, then we are faced with an issue. The only way we can have an incorrect final print out is if one of our variables had an incorrect state (infected at some time). However, the way the code is written, if the number passes through the for-loop without breaking, it is assumed a prime number. Therefore, we only enter the fault when we have a prime number that ends in 9. Because of this, there is no way to enter the fault without also propagating into the final state.
